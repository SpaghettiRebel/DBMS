# DBMS
Простая SQL СУБД на C++  
  
dbms_project/  
├── CMakeLists.txt            # Главный скрипт сборки всего проекта  
├── third_party/              # Сюда положите чужие библиотеки (json.hpp, crow.h)  
├── shared/                   # То, что нужно и клиенту, и серверу  
│   └── QueryPlan.h           # Файл-контракт между парсером и движком
│  
├── client/                   # ПРОГРАММА 1: Клиентский терминал [Разработчик 2]  
│   ├── CMakeLists.txt          
│   └── main.cpp              # Читает ввод из консоли/файла и шлет HTTP-запрос  
│  
├── server/                   # ПРОГРАММА 2: Ядро СУБД  
│   ├── CMakeLists.txt        
│   ├── main.cpp              # Точка входа: запускает HTTP-сервер  
│   │  
│   ├── network/              # [Разработчик 2] Сетевой слой  
│   │   ├── http_server.cpp   # Настройка Crow, прием запросов от клиента  
│   │   └── auth.cpp          # JWT-токены и проверка паролей
│   │  
│   ├── parser/               # [Разработчик 2] Переводчик с SQL на C++  
│   │   ├── lexer.l           # Файл для Flex (распознает слова SELECT, INSERT)  
│   │   └── grammar.y         # Файл для Bison (собирает слова в QueryPlan)  
│   │  
│   └── storage/              # [Разработчик 1] Работа с жестким диском  
│       ├── bplus_tree.cpp    # Алгоритм вашего B+-дерева    
│       ├── file_manager.cpp  # Открытие/чтение/запись файлов  
│       └── engine.cpp        # Главный класс, который берет QueryPlan и лезет в файлы  
│  
└── data/                     # Здесь СУБД будет физически создавать папки БД и файлы таблиц
